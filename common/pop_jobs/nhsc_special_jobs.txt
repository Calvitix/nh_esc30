### Crystal Farmer : produces Living Crystals and some alloys

esc_crystal_farmer = {
    category = worker
    condition_string = WORKER_JOB_TRIGGER
    building_icon = building_esc_graviton_condenser
    clothes_texture_index = 4

    possible = {
        worker_job_check_trigger = yes
    }

    resources = {
        category = miner
        produces = {
            sr_esc_living_crystal = 2
            alloys = 1
        }   
        upkeep = {
            energy = 3
            minerals = 4
        }
    }

    weight = {
        weight = 1 #@worker_job_weight
        modifier = {
            factor = 0.2
            has_citizenship_rights = no
        }
        modifier = {
            factor = 1.5
            OR = {
                has_trait = trait_strong
                has_trait = trait_very_strong
                has_trait = trait_agrarian
                has_trait = trait_industrious
                has_trait = trait_cybernetic
            }
        }
        modifier = {
            factor = 0.75
            has_trait = trait_nerve_stapled
            has_trair = trait_weak
        }
        modifier = {
            factor = 1.5
            has_job = esc_crystal_farmer
        }
    }
}
